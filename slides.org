M = MOVES

  newState = _.clone(prevState)
  switch(ev)  {
    case M.LEFT: newState.x -= 1; break;
    case M.RIGHT: newState.x += 1; break;
    ...
  }
  return newState;
}
_.reduce([M.LEFT, M.UP, M.RIGHT], reduceLocationState, initialState)
